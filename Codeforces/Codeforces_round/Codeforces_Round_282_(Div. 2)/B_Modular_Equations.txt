lil power(lil a, lil b){
	lil res =1;
	while(b>0){
		if(b%2==1)
			res = ((res%mod) * (a%mod))%mod;
		a = ((a%mod)*(a%mod))%mod;
		b /= 2;
	}
	return res%mod;
}
