#define fd(i,s,e) for(int i=s;i>e;i--)

#define aout(i,arr,n); for(int i=0;i<n;i++){ cout << arr[i] << "\t"; }

#define mod 1000000007
using namespace std;

int main(){
    lil n,k;
    cin >> n >> k;
    lil cnt=0,c=0;
    lil x=-1;
    for(lil i=1;i*i<=n;i++){
        if(n%i==0){
            c += 2;
            if(i==n/i)
            c--;
        }
    }
    for(lil i=1;i*i<=n;i++){
        if(n%i==0)
        cnt++;
        if(cnt==k){
            x=i;
            break;
        }
        else if(c-cnt+1==k){
            x=n/i;
            break;
        }
    }
    cout(x);
}
